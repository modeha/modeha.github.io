<script>
let arText = "";
let enText = "";
let surahAndAyah = "";

document.addEventListener("DOMContentLoaded", function () {
  getQuote();

  document.getElementById("shuffle").addEventListener("click", getQuote);

  document.getElementById("tweet").addEventListener("click", function () {
    if (!enText || !surahAndAyah) {
      alert("Please wait until the Ayah loads.");
      return;
    }

    const tweetText = '"' + enText + '" QS ' + surahAndAyah;
    const tweetLink = "https://twitter.com/intent/tweet?text=" + encodeURIComponent(tweetText);
    window.open(tweetLink, "_blank");
  });
});

function getAdvanced19RandomAyahNumber() {
  const totalAyahs = 6236;
  const candidates = [];

  for (let globalAyah = 1; globalAyah <= totalAyahs; globalAyah++) {
    let weight = 1;

    if (globalAyah % 19 === 0) {
      weight += 10;
    }

    if (String(globalAyah).includes("19")) {
      weight += 5;
    }

    const digitSum = String(globalAyah)
      .split("")
      .map(Number)
      .reduce((a, b) => a + b, 0);

    if (digitSum % 19 === 0) {
      weight += 4;
    }

    const remainder = globalAyah % 19;

    if (remainder === 1 || remainder === 18) {
      weight += 2;
    }

    candidates.push({
      ayah: globalAyah,
      weight: weight
    });
  }

  const totalWeight = candidates.reduce((sum, item) => sum + item.weight, 0);
  let randomValue = Math.random() * totalWeight;

  for (const item of candidates) {
    randomValue -= item.weight;

    if (randomValue <= 0) {
      return item.ayah;
    }
  }

  return Math.floor(Math.random() * totalAyahs) + 1;
}

async function getQuote() {
  document.getElementById("arabicVerseText").textContent = "Loading...";
  document.getElementById("verseText").textContent = "";
  document.getElementById("surahAndAyah").textContent = "";
  document.getElementById("randomMethod").textContent = "";
  document.getElementById("errorMessage").textContent = "";

  const ayah = getAdvanced19RandomAyahNumber();

  try {
    const arabicUrl = "https://api.alquran.cloud/v1/ayah/" + ayah + "/quran-uthmani";
    const englishUrl = "https://api.alquran.cloud/v1/ayah/" + ayah + "/en.asad";

    const arabicResponse = await fetch(arabicUrl);
    const englishResponse = await fetch(englishUrl);

    if (!arabicResponse.ok || !englishResponse.ok) {
      throw new Error("API response was not OK.");
    }

    const arabicResult = await arabicResponse.json();
    const englishResult = await englishResponse.json();

    if (!arabicResult.data || !englishResult.data) {
      throw new Error("API returned incomplete data.");
    }

    const arabicData = arabicResult.data;
    const englishData = englishResult.data;

    arText = arabicData.text;
    enText = englishData.text;

    const surah = englishData.surah.englishName;
    const ayahNumber = englishData.numberInSurah;

    surahAndAyah = surah + " : " + ayahNumber;

    document.getElementById("arabicVerseText").textContent = arText;
    document.getElementById("verseText").textContent = enText;
    document.getElementById("surahAndAyah").textContent = surahAndAyah;
    document.getElementById("randomMethod").textContent =
      "Selected using advanced 19-based weighted random method. Global Ayah: " + ayah;

  } catch (error) {
    document.getElementById("arabicVerseText").textContent = "";
    document.getElementById("verseText").textContent = "";
    document.getElementById("surahAndAyah").textContent = "";
    document.getElementById("randomMethod").textContent = "";
    document.getElementById("errorMessage").textContent =
      "The Quran API could not be reached. Please check your internet connection or browser settings.";

    console.error(error);
  }
}
